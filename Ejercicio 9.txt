import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.Select;
import org.testng.Assert;
import org.testng.annotations.Test;

import java.util.List;

public class registrationFacebookTest {

    WebDriver driver;

    @Test

    public void fullRegistrationTest() throws InterruptedException {

        WebDriver driver =  getDriver();
        driver.findElement(By.linkText("Crear cuenta nueva")).click();
        System.out.println("Link crear cuenta " + driver.getCurrentUrl());

        Thread.sleep(2000);

        WebElement firstname = driver.findElement(By.name("firstname"));
        firstname.sendKeys("John");

        WebElement lastname = driver.findElement(By.name("lastname"));
        lastname.sendKeys("Smith");

        WebElement phone = driver.findElement(By.name("reg_email__"));
        phone.sendKeys("5555555");

        WebElement password = driver.findElement(By.name("reg_passwd__"));
        password.sendKeys("123456789");

        WebElement elementoDia = driver.findElement(By.name("birthday_day"));
        Select dia = new Select(elementoDia);
        dia.selectByValue("26");

        WebElement elementoMes = driver.findElement(By.name("birthday_month"));
        Select mes = new Select(elementoMes);
        mes.selectByVisibleText("jun");

        WebElement elementoAnio = driver.findElement(By.name("birthday_year"));
        Select anio = new Select(elementoAnio);
        anio.selectByValue("1980");

        List<WebElement> listaSexos = driver.findElements(By.name("sex"));
        Assert.assertEquals(listaSexos.size(),3);

        WebElement male = listaSexos.get(1);
        male.click();

        Thread.sleep(6000);
        driver.close();

    }

    public WebDriver getDriver() {

        System.setProperty("webdriver.chrome.driver", "drivers/chromedriver.exe");
        driver = new ChromeDriver();
        driver.get("https://www.facebook.com/");
        return driver;
    }


}
